void func ::sin()
{
    double x=choice_sintime(m_x);
    double sum=0;
    int n_n =0;
    for(int n=0;n<N;n++)
      {
        int m=1;
        double x_x=x;
        for(int j=1;j<=2*n+1;j++)
          m*=j;
          n_n=pow(-1,n);
          x_x=pow(x_x,2*n+1);
          sum+=(x_x/m)*n_n;
       }
 }
